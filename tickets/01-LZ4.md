# Ticket 01: Native Tart Image Support via LZ4 Decompression

## Summary

Add native support for pulling Tart (Cirrus Labs) macOS VM images from OCI registries like `ghcr.io/cirruslabs/macos-*`. This eliminates the need for the Tart CLI as a runtime dependency by extending the existing ORAS-based OCI pipeline with LZ4 decompression and Tart media type handling.

## Background

Tart images are standard OCI artifacts with:
- **LZ4 compression** (vs. GZIP in our format)
- **Multiple chunked layers** for the disk image (vs. single blob)
- **Different media types** (Cirrus Labs namespace vs. Agoda namespace)

Both formats produce identical VM artifacts: `disk.img` and `nvram.bin` (aux.img).

## Requirements

- Pull Tart images directly: `ghcr.io/cirruslabs/macos-sequoia-xcode:16`
- No external Tart CLI dependency
- Unified cache directory (no duplicate storage)
- Same Pod spec for both image formats
- Transparent format detection

```go
const (
    // TartConfigMediaType is the media type for Tart configuration.
    TartConfigMediaType MediaType = "application/vnd.cirruslabs.tart.config.v1"

    // TartDiskMediaTypeV1 is the legacy media type for Tart disk images.
    TartDiskMediaTypeV1 MediaType = "application/vnd.cirruslabs.tart.disk.v1"

    // TartDiskMediaTypeV2 is the current media type for Tart disk images.
    TartDiskMediaTypeV2 MediaType = "application/vnd.cirruslabs.tart.disk.v2"

    // TartNVRAMMediaType is the media type for Tart NVRAM data.
    TartNVRAMMediaType MediaType = "application/vnd.cirruslabs.tart.nvram.v1"
)

// IsTartMediaType returns true if the media type is a Tart media type.
func IsTartMediaType(mediaType string) bool {
    return strings.HasPrefix(mediaType, "application/vnd.cirruslabs.tart")
}

// IsTartDiskLayer returns true if the media type is a Tart disk layer.
func IsTartDiskLayer(mediaType string) bool {
    return mediaType == string(TartDiskMediaTypeV1) || mediaType == string(TartDiskMediaTypeV2)
}

// IsTartConfig returns true if the media type is a Tart config.
func IsTartConfig(mediaType string) bool {
    return mediaType == string(TartConfigMediaType)
}
```

### Step 2: Implement LZ4 Decompressor

**File:** `pkg/oci/lz4.go` (new file)

```go
package oci

import (
    "context"
    "fmt"
    "io"
    "os"

    "github.com/pierrec/lz4/v4"
    ocispec "github.com/opencontainers/image-spec/specs-go/v1"
    "github.com/virtual-kubelet/virtual-kubelet/trace"
)

// DecompressTartDiskLayers handles Tart's multi-layer LZ4 format.
// IMPORTANT: Layers MUST be processed sequentially because Tart's LZ4 frames
// can span layer boundaries. ORAS downloads in parallel, but we decompress
// sequentially after all layers are fetched.
func (s *Store) DecompressTartDiskLayers(ctx context.Context, layers []ocispec.Descriptor, destPath string) (err error) {
    ctx, span := trace.StartSpan(ctx, "OCI.DecompressTartDiskLayers")
    defer func() {
        span.SetStatus(err)
        span.End()
    }()

    dest, err := os.Create(destPath)
    if err != nil {
        return fmt.Errorf("create disk file: %w", err)
    }
    defer func() {
        if cerr := dest.Close(); cerr != nil && err == nil {
            err = cerr
        }
    }()

    // Create pipe: layers write compressed data, LZ4 reader decompresses
    pr, pw := io.Pipe()
    lz4Reader := lz4.NewReader(pr)

    // Decompress in background goroutine
    var decompressErr error
    done := make(chan struct{})
    go func() {
        defer close(done)
        _, decompressErr = io.Copy(dest, lz4Reader)
    }()

    // Feed layers sequentially (order matters for LZ4 frame continuity)
    for i, layer := range layers {
        rc, err := s.Fetch(ctx, layer)
        if err != nil {
            pw.CloseWithError(fmt.Errorf("fetch layer %d (%s): %w", i, layer.Digest, err))
            <-done
            return fmt.Errorf("fetch layer %d: %w", i, err)
        }

        _, copyErr := io.Copy(pw, rc)
        rc.Close()
        if copyErr != nil {
            pw.CloseWithError(fmt.Errorf("copy layer %d: %w", i, copyErr))
            <-done
            return fmt.Errorf("copy layer %d: %w", i, copyErr)
        }
    }
    pw.Close()

    // Wait for decompression to complete
    <-done
    if decompressErr != nil {
        return fmt.Errorf("decompress LZ4: %w", decompressErr)
    }

    if err := dest.Sync(); err != nil {
        return fmt.Errorf("sync disk file: %w", err)
    }

    return nil
}

// DecompressSingleLZ4Layer decompresses a single LZ4-compressed layer (e.g., NVRAM).
func (s *Store) DecompressSingleLZ4Layer(ctx context.Context, layer ocispec.Descriptor, destPath string) (err error) {
    ctx, span := trace.StartSpan(ctx, "OCI.DecompressSingleLZ4Layer")
    defer func() {
        span.SetStatus(err)
        span.End()
    }()

    rc, err := s.Fetch(ctx, layer)
    if err != nil {
        return fmt.Errorf("fetch layer: %w", err)
    }
    defer rc.Close()

    dest, err := os.Create(destPath)
    if err != nil {
        return fmt.Errorf("create file: %w", err)
    }
    defer func() {
        if cerr := dest.Close(); cerr != nil && err == nil {
            err = cerr
        }
    }()

    lz4Reader := lz4.NewReader(rc)
    if _, err := io.Copy(dest, lz4Reader); err != nil {
        return fmt.Errorf("decompress LZ4: %w", err)
    }

    return dest.Sync()
}
```

```go
        DPI    int `json:"dpi"`
    } `json:"display,omitempty"`

    HardwareModel string `json:"hardwareModel"`
    // ECID is the machine identifier, base64-encoded
    ECID          string `json:"ecid"`
    MacAddress    string `json:"macAddress"`

// ToConfig converts a TartConfig to our internal Config format.
func (tc *TartConfig) ToConfig() Config {
    return Config{
        MediaType:         MediaTypeConfigV1,
        OS:                tc.OS,
        HardwareModelData: tc.HardwareModel,  // Already base64 encoded
        MachineIdData:     tc.ECID,           // Already base64 encoded
        Storage: []MediaType{
            MediaTypeAuxImage,
            MediaTypeDiskImage,
        },
    }
}
```

### Step 4: Implement Tart Image Processing

**File:** `pkg/oci/tart.go` (new file)

```go
package oci

import (
    "context"
    "encoding/json"
    "fmt"
    "path/filepath"

    ocispec "github.com/opencontainers/image-spec/specs-go/v1"
    "github.com/virtual-kubelet/virtual-kubelet/log"
    "github.com/virtual-kubelet/virtual-kubelet/trace"
)

// processTartManifest handles a Tart OCI manifest, separating disk layers
// from config and NVRAM, then decompressing appropriately.
func (s *Store) processTartManifest(ctx context.Context, manifest ocispec.Manifest) (err error) {
    ctx, span := trace.StartSpan(ctx, "OCI.processTartManifest")
    defer func() {
        span.SetStatus(err)
        span.End()
    }()
    logger := log.G(ctx)

    // Categorize layers by type
    var diskLayers []ocispec.Descriptor
    var nvramLayer *ocispec.Descriptor
    var configDesc *ocispec.Descriptor

    // Config is in manifest.Config, not layers
    if IsTartConfig(manifest.Config.MediaType) {
        configDesc = &manifest.Config
    }

    for _, layer := range manifest.Layers {
        switch {
        case IsTartDiskLayer(layer.MediaType):
            diskLayers = append(diskLayers, layer)
        case layer.MediaType == string(TartNVRAMMediaType):
            nvramLayer = &layer
        }
    }

    logger.Debugf("Tart manifest: %d disk layers, nvram=%v, config=%v",
        len(diskLayers), nvramLayer != nil, configDesc != nil)

    // Process disk layers (LZ4, sequential)
    if len(diskLayers) > 0 {
        diskPath := filepath.Join(s.workingDir, "disk.img")
        if err := s.DecompressTartDiskLayers(ctx, diskLayers, diskPath); err != nil {
            return fmt.Errorf("decompress tart disk: %w", err)
        }
        s.mediaTypeToPath.Store(string(MediaTypeDiskImage), diskPath)
        logger.Debugf("Decompressed Tart disk to %s", diskPath)
    }

    // Process NVRAM layer (single LZ4 layer)
    if nvramLayer != nil {
        nvramPath := filepath.Join(s.workingDir, "nvram.bin")
        if err := s.DecompressSingleLZ4Layer(ctx, *nvramLayer, nvramPath); err != nil {
            return fmt.Errorf("decompress tart nvram: %w", err)
        }
        // Map to our media type for consistent retrieval
        s.mediaTypeToPath.Store(string(MediaTypeAuxImage), nvramPath)
        logger.Debugf("Decompressed Tart NVRAM to %s", nvramPath)
    }

    // Process config
    if configDesc != nil {
        if err := s.processTartConfig(ctx, *configDesc); err != nil {
            return fmt.Errorf("process tart config: %w", err)
        }
    }

    return nil
}

// processTartConfig fetches and parses the Tart config, converting it to our format.
func (s *Store) processTartConfig(ctx context.Context, configDesc ocispec.Descriptor) (err error) {
    ctx, span := trace.StartSpan(ctx, "OCI.processTartConfig")
    defer func() {
        span.SetStatus(err)
        span.End()
    }()

    rc, err := s.Fetch(ctx, configDesc)
    if err != nil {
        return fmt.Errorf("fetch tart config: %w", err)
    }
    defer rc.Close()

    var tartConfig TartConfig
    if err := json.NewDecoder(rc).Decode(&tartConfig); err != nil {
        return fmt.Errorf("decode tart config: %w", err)
    }

    // Convert to our config format
    config := tartConfig.ToConfig()
    config.SourceFormat = "tart"

    // Write config to disk
    configPath := filepath.Join(s.workingDir, "config.json")
    configFile, err := os.Create(configPath)
    if err != nil {
        return fmt.Errorf("create config file: %w", err)
    }
    defer configFile.Close()

    if err := json.NewEncoder(configFile).Encode(&config); err != nil {
        return fmt.Errorf("encode config: %w", err)
    }

    s.mediaTypeToPath.Store(string(MediaTypeConfigV1), configPath)
    return nil
}

// IsTartManifest checks if a manifest is a Tart image by examining the config media type.
func IsTartManifest(manifest ocispec.Manifest) bool {
    return IsTartConfig(manifest.Config.MediaType)
}
```

**File:** `pkg/oci/oci.go`

Modify the `Push()` method to detect and handle Tart manifests. The key insight is that ORAS calls `Push()` for each layer, but we need to process the manifest as a whole for Tart images.

```go
type Store struct {

    tartManifest     *ocispec.Manifest

}
```

Add method to handle manifest detection:

```go
// HandleManifest is called after ORAS completes the copy to process Tart manifests.
// For ORAS images, processing happens during Push(). For Tart images, we need
// to process all layers together after they're downloaded.
func (s *Store) HandleManifest(ctx context.Context, manifest ocispec.Manifest) error {
    if IsTartManifest(manifest) {
        return s.processTartManifest(ctx, manifest)
    }
    return nil
}
```

```go
func pull(ctx context.Context, ref string, store *oci.Store, creds resource.RegistryCredentials) (desc *ocispec.Descriptor, err error) {
    // ... existing code up to oras.Copy() ...

    descOras, err := oras.Copy(ctx, repo, repo.Reference.Reference, store, repo.Reference.Reference, oras.DefaultCopyOptions)
    if err != nil {
        return nil, err
    }

    // Fetch and process manifest for Tart images
    manifest, err := fetchManifest(ctx, repo, descOras)
    if err != nil {
        return nil, fmt.Errorf("fetch manifest: %w", err)
    }

    if err := store.HandleManifest(ctx, manifest); err != nil {
        return nil, fmt.Errorf("handle manifest: %w", err)
    }

    return &descOras, nil
}

// fetchManifest retrieves the manifest from a repository.
func fetchManifest(ctx context.Context, repo *remote.Repository, desc ocispec.Descriptor) (ocispec.Manifest, error) {
    rc, err := repo.Fetch(ctx, desc)
    if err != nil {
        return ocispec.Manifest{}, err
    }
    defer rc.Close()

    var manifest ocispec.Manifest
    if err := json.NewDecoder(rc).Decode(&manifest); err != nil {
        return ocispec.Manifest{}, err
    }
    return manifest, nil
}
```

## Notes

### LZ4 Frame Continuity

Tart's LZ4 compression uses streaming frames that can span layer boundaries. This means:
1. Layers must be decompressed in order
2. You cannot parallelize decompression (though ORAS can parallelize download)
3. The LZ4 reader must receive a continuous stream of all layer data

### Registry Authentication

Tart images on `ghcr.io` may require authentication. The existing `imagePullSecrets` mechanism handles this — no changes needed.

### Disk Size

Tart disk images are typically 60-80GB uncompressed. Ensure adequate disk space and consider progress reporting for large downloads.
